# 🚂 Roster UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the existing roster UI with three view modes (avatar number plate, image card, list table) as shared components in `@repo/ui`, integrated into cloud and throttle apps.

**Architecture:** Four new components in `@repo/ui` — `LocoNumberPlate`, `LocoCard`, `LocoListRow`, `LocoRoster` — replacing `LocoAvatar`, cloud's `RosterItem`/`RosterList`, and throttle's `RosterView` content. Logo assets move from throttle to `@repo/ui`. `roadnameLogos.ts` moves to `@repo/ui` and is re-exported.

**Tech Stack:** Vue 3 + `<script setup lang="ts">`, Vuetify 3, Tailwind CSS, Roboto Condensed (Google Font), Vuefire

**Spec:** `docs/superpowers/specs/2026-04-03-roster-ui-design.md`

```vue
<script setup lang="ts">
import { computed } from 'vue'
import { ROADNAMES } from '@repo/modules/locos'

interface PlateSize {
  width: string
  height: string
  fontSize: string
  letterSpacing: string
  borderRadius: string
  outerPadding: string
  borderWidth: string
  rivetSize: string
  rivetOffset: string
  rivets: ('tl' | 'tr' | 'bl' | 'br' | 'tc' | 'bc')[]
}

const SIZE_MAP: Record<string, PlateSize> = {
  xs:  { width: '52px',  height: '22px', fontSize: '14px', letterSpacing: '1.5px', borderRadius: '5px',  outerPadding: '2px', borderWidth: '1.5px', rivetSize: '2px',  rivetOffset: '3px', rivets: ['tl', 'tr'] },
  sm:  { width: '76px',  height: '30px', fontSize: '18px', letterSpacing: '2px',   borderRadius: '7px',  outerPadding: '2px', borderWidth: '1.5px', rivetSize: '3px',  rivetOffset: '4px', rivets: ['tl', 'tr'] },
  md:  { width: '104px', height: '40px', fontSize: '26px', letterSpacing: '3px',   borderRadius: '9px',  outerPadding: '3px', borderWidth: '2px',   rivetSize: '4px',  rivetOffset: '5px', rivets: ['tl', 'tr'] },
  lg:  { width: '136px', height: '52px', fontSize: '34px', letterSpacing: '4px',   borderRadius: '11px', outerPadding: '3px', borderWidth: '2px',   rivetSize: '5px',  rivetOffset: '6px', rivets: ['tl', 'tr', 'bl', 'br'] },
  xl:  { width: '176px', height: '66px', fontSize: '44px', letterSpacing: '5px',   borderRadius: '14px', outerPadding: '4px', borderWidth: '2.5px', rivetSize: '5px',  rivetOffset: '8px', rivets: ['tl', 'tr', 'bl', 'br', 'tc'] },
  '2xl': { width: '220px', height: '82px', fontSize: '56px', letterSpacing: '7px', borderRadius: '16px', outerPadding: '5px', borderWidth: '3px',   rivetSize: '6px',  rivetOffset: '10px', rivets: ['tl', 'tr', 'bl', 'br', 'tc', 'bc'] },
}

const PLATE_COLORS: Record<string, { border: string; text: string }> = {
  orange:  { border: '#e87020', text: '#e87020' },
  sky:     { border: '#5080bb', text: '#80b0e8' },
  yellow:  { border: '#d4b830', text: '#e8c840' },
  red:     { border: '#c02828', text: '#e04040' },
  indigo:  { border: '#4060c0', text: '#5878d8' },
  zinc:    { border: '#777',    text: '#b0b0b0' },
  blue:    { border: '#4070aa', text: '#6090cc' },
  green:   { border: '#3a7a3a', text: '#50a850' },
}

const DEFAULT_PLATE_COLOR = { border: '#b8972e', text: '#d4b44a' }

const props = withDefaults(defineProps<{
  address: number
  color?: string
  size?: 'xs' | 'sm' | 'md' | 'lg' | 'xl' | '2xl'
  showLabel?: boolean
  label?: string
}>(), {
  size: 'md',
  showLabel: false,
})

const sizeConfig = computed(() => SIZE_MAP[props.size] || SIZE_MAP.md)

const plateColor = computed(() => {
  if (props.color && PLATE_COLORS[props.color]) return PLATE_COLORS[props.color]
  return DEFAULT_PLATE_COLOR
})

const outerStyle = computed(() => ({
  background: `linear-gradient(180deg, ${plateColor.value.border}, ${adjustBrightness(plateColor.value.border, -20)})`,
  borderRadius: sizeConfig.value.borderRadius,
  padding: sizeConfig.value.outerPadding,
  display: 'inline-block',
}))

const faceStyle = computed(() => ({
  background: '#111',
  borderRadius: `calc(${sizeConfig.value.borderRadius} - 2px)`,
  borderWidth: sizeConfig.value.borderWidth,
  borderStyle: 'solid',
  borderColor: plateColor.value.border,
  width: sizeConfig.value.width,
  height: sizeConfig.value.height,
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  position: 'relative' as const,
}))

const numStyle = computed(() => ({
  fontWeight: 700,
  fontFamily: "'Roboto Condensed', sans-serif",
  fontSize: sizeConfig.value.fontSize,
  letterSpacing: sizeConfig.value.letterSpacing,
  lineHeight: '1.2',
  color: plateColor.value.text,
}))

function rivetStyle(pos: string) {
  const s = sizeConfig.value
  const size = s.rivetSize
  const off = s.rivetOffset
  const base: Record<string, string> = {
    position: 'absolute',
    width: size,
    height: size,
    borderRadius: '50%',
    background: 'radial-gradient(circle at 40% 40%, #666, #333)',
    boxShadow: 'inset 0 1px 2px rgba(0,0,0,0.8)',
  }
  if (pos === 'tl') { base.top = off; base.left = off }
  if (pos === 'tr') { base.top = off; base.right = off }
  if (pos === 'bl') { base.bottom = off; base.left = off }
  if (pos === 'br') { base.bottom = off; base.right = off }
  if (pos === 'tc') { base.top = off; base.left = '50%'; base.transform = 'translateX(-50%)' }
  if (pos === 'bc') { base.bottom = off; base.left = '50%'; base.transform = 'translateX(-50%)' }
  return base
}

function adjustBrightness(hex: string, amount: number): string {
  const num = parseInt(hex.replace('#', ''), 16)
  const r = Math.max(0, Math.min(255, ((num >> 16) & 0xff) + amount))
  const g = Math.max(0, Math.min(255, ((num >> 8) & 0xff) + amount))
  const b = Math.max(0, Math.min(255, (num & 0xff) + amount))
  return `#${((r << 16) | (g << 8) | b).toString(16).padStart(6, '0')}`
}
</script>

<template>
  <div class="inline-flex flex-col items-center">
    <div :style="outerStyle">
      <div :style="faceStyle">
        <div v-for="pos in sizeConfig.rivets" :key="pos" :style="rivetStyle(pos)" />
        <span :style="numStyle">{{ address }}</span>
      </div>
    </div>
    <span
      v-if="showLabel && label"
      class="text-xs text-gray-400 mt-1 truncate max-w-full"
    >
      {{ label }}
    </span>
  </div>
</template>
```

<script setup lang="ts">
import { computed } from 'vue'
import type { Loco } from '@repo/modules/locos'
import { ROADNAMES } from '@repo/modules/locos'
import LocoNumberPlate from './LocoNumberPlate.vue'
import { getRoadnameMedia } from './roadnameLogos'

const props = defineProps<{
  loco: Loco
}>()

const emit = defineEmits<{
  click: [loco: Loco]
}>()

const roadname = computed(() =>
  props.loco.meta?.roadname
    ? ROADNAMES.find((r) => r.value === props.loco.meta?.roadname)
    : null
)

const media = computed(() => getRoadnameMedia(props.loco.meta?.roadname))
const hasImage = computed(() => !!props.loco.meta?.imageUrl)
const consistCount = computed(() => props.loco.consist?.length ?? 0)
const plateColor = computed(() => roadname.value?.color || 'yellow')
</script>

<template>
  <v-card
    class="cursor-pointer overflow-hidden"
    rounded="lg"
    variant="flat"
    color="surface-variant"
    @click="emit('click', loco)"
  >
    <!-- Image area -->
    <div class="relative h-40 flex items-center justify-center overflow-hidden" :class="!hasImage ? media?.fallbackClass || 'bg-gray-700' : ''">
      <!-- User photo -->
      <img
        v-if="hasImage"
        :src="loco.meta!.imageUrl"
        :alt="loco.name"
        class="w-full h-full object-cover"
      />
      <!-- Fallback: roadname logo watermark -->
      <img
        v-else-if="media?.logo"
        :src="media.logo"
        :alt="roadname?.label || ''"
        class="h-20 opacity-30 object-contain"
      />

      <!-- Consist badge -->
      <v-chip
        v-if="consistCount > 0"
        size="x-small"
        color="pink"
        variant="flat"
        class="absolute top-2 right-2"
      >
        consist · {{ consistCount }}
      </v-chip>

      <!-- Number plate overlay -->
      <div class="absolute bottom-2 right-2">
        <LocoNumberPlate :address="loco.address" :color="plateColor" size="sm" />
      </div>
    </div>

    <!-- Info strip -->
    <v-card-text class="pa-3">
      <div class="text-body-2 font-weight-medium">{{ loco.name || `Loco ${loco.address}` }}</div>
      <div class="text-caption text-medium-emphasis">
        {{ roadname?.label || 'Unknown' }} · #{{ loco.address }}
      </div>
    </v-card-text>
  </v-card>
</template>
